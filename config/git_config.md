# Git Configurations
```
git config --global user.name "John Doe"                # config user name
git config --global user.email johndoe@example.com      # config user email
git config --global core.editor vim                     # config editor
git config --global credential.helper cache             # saving password

### command alias setting
git config --global alias.st status 
git config --global alias.ci commit
git config --global alias.cm "commit -m"
git config --global alias.co checkout
git config --global alias.br branch
git config --global alias.last "log -1 HEAD"
git config --global alias.aa "add --all"
git config --global alias.cb "checkout -b"
```
